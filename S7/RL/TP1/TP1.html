2.

n <- 100
x <- rnorm(n, 3, 1)

3.

sigma2e <- c(0.1, 0.5, 2, 6)
b0 <- 1
b1 <- 2

y <- matrix(0, ncol = 4, nrow = n)
for (i in seq_len(ncol(y)))
y[, i] <- b0 + b1 * x + rnorm(n, 0, sqrt(sigma2e[i]))

4.

par(mfrow = c(2, 2))
for (i in seq_len(4))
plot(x, y[, i], xlab = "x", ylab = "y", main = paste("sigma2e =", sigma2e[i]))
title("Nuages de points")
